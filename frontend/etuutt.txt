    function getCode(url:string) {
        const index = url.indexOf('code=');
        if (index !== -1) {
            const codeStart = index + 'code='.length;
            const code = url.substring(codeStart).split('&')[0];
            return code;
        }
        return null;
    }
    
    const code = getCode(window.location.href);
    console.log("code =" +code);
